<script>
    const KEY_NEWS = 'dussault_news_v1';
    const KEY_PROFILE = 'dussault_profile_v2';
    let newsData = [];
    let currentUserName = "Direction";

    function init() {
        // Date du jour
        const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        document.getElementById('todayDate').textContent = new Date().toLocaleDateString('fr-CA', options);

        // Nom de l'utilisateur (depuis les param√®tres)
        const profile = localStorage.getItem(KEY_PROFILE);
        if (profile) {
            const data = JSON.parse(profile);
            if (data.name) {
                currentUserName = data.name;
                document.getElementById('greeting').textContent = "Bonjour " + data.name.split(' ')[0] + " !";
            }
        }

        // Charger les nouvelles
        const stored = localStorage.getItem(KEY_NEWS);
        if (stored) {
            newsData = JSON.parse(stored);
        } else {
            // Message de bienvenue par d√©faut
            newsData = [{
                id: 'news-init',
                type: 'info',
                title: 'Bienvenue sur votre nouvelle application !',
                body: 'Toutes les nouvelles de l\'entreprise appara√Ætront ici. Vous pouvez publier des rappels de s√©curit√©, des infos sur les chantiers ou des messages pour l\'√©quipe.',
                author: 'Syst√®me',
                date: new Date().toISOString()
            }];
            localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
        }
        renderNews();
    }

    function renderNews() {
        const container = document.getElementById('newsList');
        container.innerHTML = '';

        if (newsData.length === 0) {
            container.innerHTML = '<div style="color:#888; font-style:italic; text-align:center; padding: 20px;">Aucune nouvelle pour le moment.</div>';
            return;
        }

        // Trier de la plus r√©cente √† la plus ancienne
        const sorted = [...newsData].sort((a, b) => new Date(b.date) - new Date(a.date));

        sorted.forEach(news => {
            const dateObj = new Date(news.date);
            const dateStr = dateObj.toLocaleDateString('fr-CA') + ' √† ' + dateObj.toLocaleTimeString('fr-CA', {hour: '2-digit', minute:'2-digit'});
            
            const div = document.createElement('div');
            div.className = `news-card ${news.type}`;
            div.innerHTML = `
                <div class="news-header">
                    <h3 class="news-title">${news.title}</h3>
                    <div class="news-date">${dateStr}</div>
                </div>
                <div class="news-body">${news.body}</div>
                <div class="news-author">Publi√© par : ${news.author}</div>
                <button class="btn-delete-news" onclick="deleteNews('${news.id}')" title="Supprimer">üóëÔ∏è</button>
            `;
            container.appendChild(div);
        });
    }

    function openNewsModal() {
        document.getElementById('newsTitle').value = '';
        document.getElementById('newsBody').value = '';
        document.getElementById('newsType').value = 'info';
        document.getElementById('newsModal').style.display = 'flex';
    }

    function closeNewsModal() {
        document.getElementById('newsModal').style.display = 'none';
    }

    function saveNews() {
        const title = document.getElementById('newsTitle').value.trim();
        const body = document.getElementById('newsBody').value.trim();
        const type = document.getElementById('newsType').value;

        if (!title || !body) {
            alert("Veuillez remplir le titre et le message.");
            return;
        }

        const newEntry = {
            id: 'news-' + Date.now(),
            type: type,
            title: title,
            body: body,
            author: currentUserName,
            date: new Date().toISOString()
        };

        newsData.push(newEntry);
        localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
        renderNews();
        closeNewsModal();
    }

    function deleteNews(id) {
        if(confirm("Supprimer cette annonce ?")) {
            newsData = newsData.filter(n => n.id !== id);
            localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
            renderNews();
        }
    }

    init();
</script>
