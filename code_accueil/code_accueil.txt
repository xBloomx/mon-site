<script>
    const KEY_NEWS = 'dussault_news_v1';
    const KEY_PROFILE = 'dussault_profile_v2';
    const KEY_FORMATIONS = 'dussault_formations_v1';
    let newsData = [];
    let currentUserName = "Direction";

    // --- LE CORRECTIF POUR LES BOUTONS ACTIONS RAPIDES ---
    function goToTab(targetId) {
        try {
            if (window.parent && window.parent.document) {
                const btn = window.parent.document.querySelector(`.nav-btn[data-target="${targetId}"]`);
                if (btn) {
                    btn.click();
                } else {
                    console.log("Bouton de navigation introuvable.");
                }
            }
        } catch (e) {
            console.log("Navigation bloquée par le navigateur local.");
        }
    }

    function init() {
        // Formater la date en français complet
        const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        let dateStr = new Date().toLocaleDateString('fr-CA', options);
        // Mettre la première lettre en majuscule
        dateStr = dateStr.charAt(0).toUpperCase() + dateStr.slice(1);
        document.getElementById('todayDate').textContent = dateStr;
        
        const profile = localStorage.getItem(KEY_PROFILE);
        if (profile) {
            const data = JSON.parse(profile);
            if (data.name) {
                currentUserName = data.name;
                document.getElementById('greeting').textContent = "Bonjour " + data.name.split(' ')[0];
            }
        }
        
        loadNews();
        checkCertifications();
    }

    // --- GESTION DES ALERTES ---
    function checkCertifications() {
        const container = document.getElementById('alertsList');
        container.innerHTML = '';
        const storedFormations = localStorage.getItem(KEY_FORMATIONS);
        let hasAlerts = false;

        if (storedFormations) {
            const formations = JSON.parse(storedFormations);
            const today = new Date();

            formations.forEach(f => {
                const expDate = new Date(f.dateExp);
                const diffTime = expDate - today;
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

                if (diffDays < 0) {
                    hasAlerts = true;
                    container.innerHTML += `
                        <div class="alert-item alert-danger">
                            <span class="alert-icon">⚠️</span>
                            <div><b>${f.name}</b> expirée !</div>
                        </div>`;
                } else if (diffDays <= 30) {
                    hasAlerts = true;
                    container.innerHTML += `
                        <div class="alert-item alert-warning">
                            <span class="alert-icon">⏳</span>
                            <div><b>${f.name}</b> expire dans ${diffDays}j.</div>
                        </div>`;
                }
            });
        }

        if (!hasAlerts) {
            container.innerHTML = `
                <div class="alert-item alert-success">
                    <span class="alert-icon">✅</span>
                    <div>Toutes vos cartes sont à jour.</div>
                </div>`;
        }
    }

    // --- GESTION DES NOUVELLES ---
    function loadNews() {
        const stored = localStorage.getItem(KEY_NEWS);
        if (stored) {
            newsData = JSON.parse(stored);
        } else {
            newsData = [{
                id: 'news-init', type: 'info', title: 'Bienvenue sur votre application !',
                body: 'Utilisez cet espace pour partager des informations avec votre équipe. Le menu de gauche vous permet de naviguer dans l\'application.', author: 'Système', date: new Date().toISOString()
            }];
            localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
        }
        renderNews();
    }

    function renderNews() {
        const container = document.getElementById('newsList');
        container.innerHTML = '';

        if (newsData.length === 0) {
            container.innerHTML = '<div style="background-color: var(--card-bg); border-radius:12px; border: 1px dashed #444; color:#888; text-align:center; padding: 30px;">Aucun message affiché.</div>';
            return;
        }

        const sorted = [...newsData].sort((a, b) => new Date(b.date) - new Date(a.date));

        sorted.forEach(news => {
            const dateObj = new Date(news.date);
            const dateStr = dateObj.toLocaleDateString('fr-CA') + ' - ' + dateObj.toLocaleTimeString('fr-CA', {hour: '2-digit', minute:'2-digit'});
            
            const div = document.createElement('div');
            div.className = `news-card ${news.type}`;
            div.innerHTML = `
                <div class="news-header">
                    <h3 class="news-title">${news.title}</h3>
                    <div class="news-date">${dateStr}</div>
                </div>
                <div class="news-body">${news.body}</div>
                <div class="news-author">Message de : ${news.author}</div>
                <button class="btn-delete-news" onclick="deleteNews('${news.id}')" title="Supprimer">🗑️</button>
            `;
            container.appendChild(div);
        });
    }

    function openNewsModal() {
        document.getElementById('newsTitle').value = '';
        document.getElementById('newsBody').value = '';
        document.getElementById('newsType').value = 'info';
        document.getElementById('newsModal').style.display = 'flex';
    }
    function closeNewsModal() { document.getElementById('newsModal').style.display = 'none'; }

    function saveNews() {
        const title = document.getElementById('newsTitle').value.trim();
        const body = document.getElementById('newsBody').value.trim();
        const type = document.getElementById('newsType').value;

        if (!title || !body) { alert("Veuillez remplir le titre et le message."); return; }

        const newEntry = { id: 'news-' + Date.now(), type: type, title: title, body: body, author: currentUserName, date: new Date().toISOString() };

        newsData.push(newEntry);
        localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
        renderNews();
        closeNewsModal();
    }

    function deleteNews(id) {
        if(confirm("Supprimer ce message ?")) {
            newsData = newsData.filter(n => n.id !== id);
            localStorage.setItem(KEY_NEWS, JSON.stringify(newsData));
            renderNews();
        }
    }

    init();
</script>
