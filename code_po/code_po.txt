<script>
    const KEY_POS = 'dussault_po_v1';
    let poLog = [];
    let poToDeleteId = null; 

    // --- GESTION DES MODALES G√âN√âRIQUES ---
    function showAlert(msg) {
        document.getElementById('alertMsg').textContent = msg;
        document.getElementById('alertModal').style.display = 'flex';
    }
    function closeAlertModal() { document.getElementById('alertModal').style.display = 'none'; }
    function closeModal(id) { document.getElementById(id).style.display = 'none'; }

    // --- LOGIQUE METIER (PO) ---
    function init() {
        const stored = localStorage.getItem(KEY_POS);
        if (stored) {
            poLog = JSON.parse(stored);
        } else {
            // Donn√©es de d√©monstration
            poLog = [
                { id: '1', poNumber: 'PO-260225-8452', plumber: 'Fran√ßois Dussault', supplier: 'Desch√™nes', address: '456 Rue St-Pierre', date: '2026-02-25', timestamp: Date.now() - 100000 },
                { id: '2', poNumber: 'PO-260226-1024', plumber: 'Patrick Baker', supplier: 'Wolseley', address: 'Entrep√¥t', date: '2026-02-26', timestamp: Date.now() }
            ];
            saveData();
        }
        renderPOList();
    }

    function saveData() { 
        localStorage.setItem(KEY_POS, JSON.stringify(poLog));
        renderPOList(); 
    }

    function formatDateFR(isoDate) {
        if(!isoDate) return '';
        const parts = isoDate.split('-'); 
        return `${parts[2]}/${parts[1]}/${parts[0]}`; 
    }

    function renderPOList(list = poLog) {
        const container = document.getElementById('poListContainer');
        container.innerHTML = '';
        
        // Trie par date de cr√©ation d√©croissante
        list.sort((a,b) => b.timestamp - a.timestamp);

        if (list.length === 0) {
            container.innerHTML = '<div style="color:#888; font-style:italic; padding:10px;">Aucun bon de commande g√©n√©r√© pour le moment.</div>';
            return;
        }

        list.forEach(po => {
            const card = document.createElement('div');
            card.className = `po-item`;
            
            card.innerHTML = `
                <div class="po-info">
                    <div class="po-number">${po.poNumber}</div>
                    <div class="po-details">
                        <span>üë§ <strong>${po.plumber}</strong></span>
                        <span>üè¢ ${po.supplier}</span>
                    </div>
                    <div class="po-subtext">üìç ${po.address} | üìÖ ${formatDateFR(po.date)}</div>
                </div>
                <div class="po-actions">
                    <button class="btn-copy" onclick="copyTextToClipboard('${po.poNumber}', this)">üìã Copier</button>
                    <button class="btn-delete" onclick="askDeletePO('${po.id}')" title="Supprimer">üóëÔ∏è</button>
                </div>
            `;
            container.appendChild(card);
        });
    }

    function filterPO() {
        const term = document.getElementById('poSearch').value.toLowerCase();
        const filtered = poLog.filter(po => 
            (po.poNumber || "").toLowerCase().includes(term) || 
            (po.plumber || "").toLowerCase().includes(term) || 
            (po.supplier || "").toLowerCase().includes(term) || 
            (po.address || "").toLowerCase().includes(term)
        );
        renderPOList(filtered);
    }

    // --- GENERATION DU PO ---
    function openPoModal() {
        // Reset le formulaire
        document.getElementById('inpPlombier').value = '';
        document.getElementById('inpFournisseur').value = '';
        document.getElementById('inpAdresse').value = '';
        document.getElementById('inpDate').value = new Date().toISOString().split('T')[0];
        
        // Cache la zone de r√©sultat, affiche le bouton g√©n√©rer
        document.getElementById('poResultZone').style.display = 'none';
        document.querySelector('.btn-custom-submit').style.display = 'block';
        
        // Ouvre la modal
        document.getElementById('poModal').style.display = 'flex';
    }

    function generatePONumber(dateString) {
        // Format: PO-AAMMJJ-XXXX (ex: PO-260227-1452)
        const d = new Date(dateString);
        const yy = String(d.getFullYear()).slice(-2);
        const mm = String(d.getMonth() + 1).padStart(2, '0');
        const dd = String(d.getDate()).padStart(2, '0');
        const random = Math.floor(1000 + Math.random() * 9000); // 4 chiffres al√©atoires
        return `PO-${yy}${mm}${dd}-${random}`;
    }

    function generatePO() {
        const plumber = document.getElementById('inpPlombier').value.trim();
        const supplier = document.getElementById('inpFournisseur').value.trim();
        const address = document.getElementById('inpAdresse').value.trim();
        const dateReq = document.getElementById('inpDate').value;
        
        if (!plumber || !supplier || !address || !dateReq) { 
            showAlert("‚ö†Ô∏è Merci de remplir tous les champs pour g√©n√©rer le PO."); 
            return; 
        }

        // Cr√©ation du num√©ro
        const newPoNumber = generatePONumber(dateReq);

        const newPo = {
            id: 'po-'+Date.now(),
            poNumber: newPoNumber,
            plumber: plumber,
            supplier: supplier,
            address: address,
            date: dateReq,
            timestamp: Date.now()
        };

        // Sauvegarde
        poLog.push(newPo);
        saveData();

        // Modification de l'interface visuelle (Animation)
        document.querySelector('.btn-custom-submit').style.display = 'none'; // Cache le bouton jaune
        document.getElementById('displayNewPoNumber').textContent = newPoNumber;
        
        // Reset l'√©tat du bouton copier au cas o√π
        const btnCopy = document.getElementById('btnCopyMain');
        btnCopy.innerHTML = "üìã Copier le num√©ro";
        btnCopy.className = "btn-copy-large";

        document.getElementById('poResultZone').style.display = 'block'; // Affiche la boite noire
    }

    // --- FONCTION DE COPIE ---
    function copyPoNumber(elementId, btnElement) {
        const text = document.getElementById(elementId).textContent;
        copyTextToClipboard(text, btnElement, true);
    }

    function copyTextToClipboard(text, btnElement, isLargeBtn = false) {
        navigator.clipboard.writeText(text).then(() => {
            // Animation du bouton pour confirmer la copie
            const originalText = btnElement.innerHTML;
            const originalClass = btnElement.className;
            
            btnElement.innerHTML = "‚úÖ Copi√© !";
            btnElement.className = isLargeBtn ? "btn-copy-large success" : "btn-copy success";
            btnElement.style.background = "var(--btn-green)";
            btnElement.style.color = "white";

            setTimeout(() => {
                btnElement.innerHTML = originalText;
                btnElement.className = originalClass;
                btnElement.style.background = ""; // Remet le CSS d'origine
            }, 2000);
        }).catch(err => {
            console.error('Erreur lors de la copie', err);
            showAlert("Erreur : Impossible de copier le texte.");
        });
    }

    // --- SUPPRESSION ---
    function askDeletePO(id) {
        poToDeleteId = id;
        document.getElementById('confirmModal').style.display = 'flex';
    }

    function executeDeletePO() {
        if (!poToDeleteId) return;
        poLog = poLog.filter(po => po.id !== poToDeleteId);
        saveData();
        document.getElementById('confirmModal').style.display = 'none';
        poToDeleteId = null;
    }

    init();
</script>
