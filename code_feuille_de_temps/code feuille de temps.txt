<script>
    let timesheetsData = [];
    const STORAGE_KEY = 'dussault_timesheets_v1';
    let currentSheetId = null;

    // --- GESTION DES MODALES ---
    let confirmCallback = null;

    function showConfirm(msg, callback) {
        document.getElementById('confirmMsg').textContent = msg;
        confirmCallback = callback;
        document.getElementById('confirmModal').style.display = 'flex';
    }

    function closeConfirmModal() {
        document.getElementById('confirmModal').style.display = 'none';
        confirmCallback = null;
    }

    document.getElementById('confirmYesBtn').onclick = function() {
        if (confirmCallback) confirmCallback();
        closeConfirmModal();
    };

    function showAlert(msg) {
        document.getElementById('alertMsg').textContent = msg;
        document.getElementById('alertModal').style.display = 'flex';
    }

    function closeAlertModal() {
        document.getElementById('alertModal').style.display = 'none';
    }

    // --- APP LOGIC ---

    function loadData() {
        const stored = localStorage.getItem(STORAGE_KEY);
        if (stored) {
            timesheetsData = JSON.parse(stored);
        } else {
            timesheetsData = [];
        }
        renderTimesheetList(timesheetsData);
    }
    
    function saveData() { 
        localStorage.setItem(STORAGE_KEY, JSON.stringify(timesheetsData)); 
        renderTimesheetList(timesheetsData); 
    }

    const viewDashboard = document.getElementById('view-dashboard');
    const viewEditor = document.getElementById('view-editor');
    const listContainer = document.getElementById('timesheetListContainer');
    const wrapper = document.getElementById('zoom-wrapper');

    function showDashboard() { 
        viewDashboard.style.display = 'flex'; 
        viewEditor.style.display = 'none'; 
        renderTimesheetList(timesheetsData);
    }
    
    function askGoBack() { 
        showConfirm("Retourner au menu ? Assurez-vous d'avoir sauvegard√©.", function() {
            showDashboard();
        });
    }

    function openNewTimesheet() {
        currentSheetId = null;
        viewDashboard.style.display = 'none';
        viewEditor.style.display = 'flex';
        wrapper.innerHTML = '';
        addPage(); 
        resetZoom();
    }

    function openExistingTimesheet(id) {
        const sheet = timesheetsData.find(s => s.id === id);
        if (!sheet) return;
        
        currentSheetId = id;
        viewDashboard.style.display = 'none';
        viewEditor.style.display = 'flex';
        wrapper.innerHTML = '';

        sheet.pagesData.forEach(pageData => {
            const newPage = createTimePageHTML();
            wrapper.appendChild(newPage);
            setupInputLogic(newPage);
            
            const inputs = newPage.querySelectorAll('input');
            pageData.forEach((val, idx) => {
                if(inputs[idx]) inputs[idx].value = val;
            });
        });
        resetZoom();
    }

    function saveCurrentTimesheet() {
        const firstPage = wrapper.querySelector('.page');
        if(!firstPage) return;
        
        const nameInput = firstPage.querySelector('.input-nom');
        const weekInput = firstPage.querySelector('.input-semaine-debut');
        const weekEndInput = firstPage.querySelector('.input-semaine-fin');
        
        const empName = nameInput ? nameInput.value.trim() : "Employ√© Inconnu";
        const dateStr = (weekInput && weekEndInput) 
            ? `Du ${weekInput.value} au ${weekEndInput.value}` 
            : "Date Inconnue";
        
        const allPagesData = [];
        wrapper.querySelectorAll('.page').forEach(page => {
            const pageInputs = Array.from(page.querySelectorAll('input')).map(i => i.value);
            allPagesData.push(pageInputs);
        });

        const sheetId = currentSheetId || 'TS-' + Date.now();
        
        const newSheetObj = {
            id: sheetId,
            client: empName, 
            date: dateStr,
            pagesData: allPagesData,
            timestamp: Date.now()
        };

        if (currentSheetId) {
            const index = timesheetsData.findIndex(s => s.id === currentSheetId);
            if (index !== -1) timesheetsData[index] = newSheetObj;
        } else {
            timesheetsData.push(newSheetObj);
            currentSheetId = sheetId;
        }

        saveData();
        showAlert("Feuille de temps sauvegard√©e !");
    }

    function deleteTimesheet(id, event) {
        event.stopPropagation();
        showConfirm("Supprimer cette feuille de temps ?", function() {
            timesheetsData = timesheetsData.filter(s => s.id !== id);
            saveData();
        });
    }

    function renderTimesheetList(data) {
        listContainer.innerHTML = '';
        const sortedData = [...data].sort((a, b) => b.timestamp - a.timestamp);
        
        if(sortedData.length === 0) { 
            listContainer.innerHTML = '<div style="color:#888; text-align:center; padding:20px;">Aucune feuille de temps.</div>';
            return; 
        }

        sortedData.forEach(sheet => {
            const item = document.createElement('div');
            item.className = 'invoice-item';
            item.onclick = function() { openExistingTimesheet(sheet.id); };
            
            item.innerHTML = `
                <div class="inv-info">
                    <div class="inv-id">${sheet.date}</div>
                    <div class="inv-client">${sheet.client}</div>
                    <div class="inv-date" style="font-size:12px">Modifi√©: ${new Date(sheet.timestamp).toLocaleDateString()}</div>
                </div>
                <div class="inv-actions">
                    <button class="btn-icon btn-delete" onclick="deleteTimesheet('${sheet.id}', event)" title="Supprimer">üóëÔ∏è</button>
                </div>
            `;
            listContainer.appendChild(item);
        });
    }

    function filterTimesheets() {
        const searchText = document.getElementById('searchInput').value.toLowerCase();
        const filtered = timesheetsData.filter(sheet => {
            return (sheet.client.toLowerCase().includes(searchText) || 
                    sheet.date.toLowerCase().includes(searchText));
        });
        renderTimesheetList(filtered);
    }

    function createTimePageHTML() {

                <div class="time-row">
                    <span style="white-space:nowrap">Nom de l'employ√©:</span>

                </div>
                
                <div class="time-row row-semaine">
                    <span style="white-space:nowrap">Semaine du:</span>

    }

    function handleDateBlur(e) {
        const input = e.target;
        let val = input.value.trim();
        val = val.replace(/[^0-9\/\-]/g, '');
        if(!val) return;
        val = val.replace(/[\-\s]/g, '/');
        if (/^\d+$/.test(val)) {
            if (val.length === 4) { 
                val = val.substring(0,2) + '/' + val.substring(2,4) + '/' + new Date().getFullYear();
            } else if (val.length === 6) {
                val = val.substring(0,2) + '/' + val.substring(2,4) + '/20' + val.substring(4,6);
            } else if (val.length === 8) {
                val = val.substring(0,2) + '/' + val.substring(2,4) + '/' + val.substring(4,8);
            }
        }
        const parts = val.split('/');
        if (parts.length === 2) {
            val = `${parts[0].padStart(2, '0')}/${parts[1].padStart(2, '0')}/${new Date().getFullYear()}`;
        } else if (parts.length === 3) {
            if (parts[2].length === 2) parts[2] = '20' + parts[2];
            val = `${parts[0].padStart(2, '0')}/${parts[1].padStart(2, '0')}/${parts[2]}`;
        }
        input.value = val;
    }

    function setupInputLogic(pageElement) {
        const inputs = Array.from(pageElement.querySelectorAll('input'));
        inputs.forEach((input, index) => {
            
            if (input.classList.contains('input-semaine-debut') || 
                input.classList.contains('input-semaine-fin')) {
                input.addEventListener('blur', handleDateBlur);
            }

            input.addEventListener('keydown', function(e) {
                if (e.key === 'Enter') {
                    e.preventDefault();
                    const td = input.closest('td');
                    if (td) {
                        const tr = td.closest('tr');
                        const colIndex = Array.from(tr.children).indexOf(td);
                        const nextTr = tr.nextElementSibling;
                        if (nextTr) {
                            const nextTd = nextTr.children[colIndex];
                            const nextInput = nextTd ? nextTd.querySelector('input') : null;
                            if (nextInput) { nextInput.focus(); return; }
                        }
                    }
                    if (index + 1 < inputs.length) inputs[index + 1].focus();
                } else if (e.key === 'Backspace' && input.value === '') {
                    if (index - 1 >= 0) inputs[index - 1].focus();
                }
            });

            input.addEventListener('paste', function(e) {
                e.preventDefault();
                const clipboardData = (e.clipboardData || window.clipboardData).getData('text');
                const rows = clipboardData.split(/\r\n|\n|\r/);
                const currentTd = input.closest('td');
                if (!currentTd) { input.value = clipboardData.trim(); return; }
                const currentTr = currentTd.closest('tr');
                const table = currentTr.closest('table');
                const startRowIndex = Array.from(table.querySelectorAll('tbody tr')).indexOf(currentTr);
                const startColIndex = Array.from(currentTr.children).indexOf(currentTd);
                const tableRows = table.querySelectorAll('tbody tr');
                
                rows.forEach((rowText, i) => {
                    if (startRowIndex + i < tableRows.length && rowText.trim() !== "") {
                        const cells = rowText.split('\t');
                        const targetRow = tableRows[startRowIndex + i];
                        cells.forEach((cellText, j) => {
                            if (targetRow.children[startColIndex + j]) {
                                const targetInput = targetRow.children[startColIndex + j].querySelector('input');
                                if (targetInput) targetInput.value = cellText.trim();
                            }
                        });
                    }
                });
            });
        });
    }

    function duplicatePage() {
        const pages = wrapper.querySelectorAll('.page.time-style');
        if (pages.length === 0) return;
        const sourcePage = pages[pages.length - 1];
        const newPage = createTimePageHTML();
        const sourceInputs = sourcePage.querySelectorAll('input');
        const newInputs = newPage.querySelectorAll('input');
        sourceInputs.forEach((input, index) => { if(newInputs[index]) newInputs[index].value = input.value; });
        wrapper.appendChild(newPage);
        setupInputLogic(newPage);
    }

    let currentZoom = 1.0;

    function updateZoom() { wrapper.style.transform = `scale(${currentZoom})`; zoomDisplay.textContent = Math.round(currentZoom * 100) + '%'; }
    function adjustZoom(delta) { currentZoom += delta; if(currentZoom<0.3) currentZoom=0.3; if(currentZoom>3.0) currentZoom=3.0; updateZoom(); }
    function resetZoom() { currentZoom = 1.0; updateZoom(); }
    
    let initialDistance = 0, initialZoomState = 1.0;
    const scrollArea = document.getElementById('scrollArea');
    scrollArea.addEventListener('touchstart', function(e) { if (e.touches.length === 2) { initialDistance = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY); initialZoomState = currentZoom; } }, { passive: false });
    scrollArea.addEventListener('touchmove', function(e) { if (e.touches.length === 2) { e.preventDefault(); const currentDistance = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY); if (initialDistance > 0) { const diff = currentDistance / initialDistance; currentZoom = initialZoomState * diff; updateZoom(); } } }, { passive: false });
