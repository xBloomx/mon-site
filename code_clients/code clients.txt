    function init() {
        const stored = localStorage.getItem(KEY_CLIENTS);
        if (stored) clients = JSON.parse(stored);
        else { clients = DEMO_CLIENTS; saveClientsData(); }
        renderClients();
    }

    function saveClientsData() { 
        localStorage.setItem(KEY_CLIENTS, JSON.stringify(clients)); 
        renderClients(); 
    }

    function getMapsUrl(address) { return `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(address)}`; }
    
    // Modification: Fonction pour Apple Maps
    function getAppleMapsUrl(address) { return `http://maps.apple.com/?q=${encodeURIComponent(address)}`; }

    function renderClients(list = clients) {
        const container = document.getElementById('clientListContainer');
        container.innerHTML = '';
        list.sort((a,b) => (a.company || a.contact || "").localeCompare(b.company || b.contact || ""));
        
        list.forEach(c => {
            const card = document.createElement('div');
            card.className = `client-card status-${c.status}`;
            card.onclick = () => openViewModal(c.id);
            let mainContact = c.contacts[0] || { phone: '---' };
            card.innerHTML = `
                <div class="cli-info">
                    <h3>
                        ${c.status==='commercial'?'üè¢':'üë§'} ${c.company || c.contact} 
                        <span class="tag">${c.status}</span>
                        ${c.alert ? '<span class="cli-alert-badge">‚ö†Ô∏è ALERTE</span>' : ''}
                    </h3>
                    <p>üìû ${mainContact.phone || '---'}</p>
                    <p>üìç ${c.address || 'Adresse inconnue'}</p>
                </div>
                <div class="cli-actions">
                    <button class="btn-gear" onclick="openEditModal('${c.id}', event)">‚öôÔ∏è</button>
                </div>`;
            container.appendChild(card);
        });
    }

    function filterClients() {
        const term = document.getElementById('clientSearch').value.toLowerCase();
        const filtered = clients.filter(c => 
            (c.company || "").toLowerCase().includes(term) || 
            (c.contact || "").toLowerCase().includes(term) || 
            (c.address || "").toLowerCase().includes(term) ||
            (c.contacts || []).some(ct => (ct.phone || "").includes(term))
        );
        renderClients(filtered);
    }

    function toggleTarifField() {
        const status = document.getElementById('edStatus').value;
        document.getElementById('grpTarif').style.display = (status === 'commercial') ? 'block' : 'none';
    }

    function openViewModal(id) {
        currentViewingId = id;
        const c = clients.find(cl => cl.id === id);
        if(!c) return;
        
        document.getElementById('vCompany').textContent = c.company || '-';
        document.getElementById('vContact').textContent = c.contact || '-';
        document.getElementById('vStatus').textContent = c.status;
        document.getElementById('vAddress').textContent = c.address || '-';
        document.getElementById('vNotes').textContent = c.notes || 'Aucune note.';
        
        const mapsLink = getMapsUrl(c.address);
        const appleLink = getAppleMapsUrl(c.address); // Nouveau lien Apple
        
        document.getElementById('vLinkGoogle').href = mapsLink;
        document.getElementById('vLinkApple').href = appleLink; // Affectation du lien

        const alertSec = document.getElementById('vAlertSection');
        if(c.alert) { alertSec.style.display = 'block'; document.getElementById('vAlertText').textContent = c.alert; }
        else { alertSec.style.display = 'none'; }

        const secTarif = document.getElementById('vTarifSection');
        if(c.status === 'commercial' && c.tarif) { secTarif.style.display = 'block'; document.getElementById('vTarif').textContent = c.tarif; }
        else { secTarif.style.display = 'none'; }

        contList.innerHTML = '';
        (c.contacts || []).forEach(ct => {
            contList.innerHTML += `
            <div class="contact-item">
                <div><span class="ci-name">${ct.name}</span> ‚Ä¢ <span class="ci-role">${ct.role}</span></div>
                <div class="ci-details"><div>${ct.phone}</div><div style="opacity:0.6; font-size:11px;">${ct.email || ''}</div></div>
            </div>`;
        });
        document.getElementById('viewModal').style.display = 'flex';
    }

    function saveClient() {
        const id = document.getElementById('edId').value;
        const alertVal = document.getElementById('edAlert').value.trim();
        const company = document.getElementById('edCompany').value.trim();
        const contact = document.getElementById('edContact').value.trim();
        const address = document.getElementById('edAddress').value.trim();
        const status = document.getElementById('edStatus').value;
        const notes = document.getElementById('edNotes').value;
        const tarif = document.getElementById('edTarif').value;
        
        // Validation Nom
        if (!company && !contact) {
            openAlertModal("Vous devez entrer au moins un Nom d'entreprise ou un Contact principal.");
            return; 
        }

        // Validation Adresse
        if (!address) {
            openAlertModal("L'adresse est obligatoire pour cr√©er la fiche client.");
            return;
        }

        // Validation Contacts
        let contacts = [];
        let hasPhone = false;

        document.querySelectorAll('.contact-row').forEach(row => {
            const name = row.querySelector('.inp-name').value.trim();
            const role = row.querySelector('.inp-role').value.trim();
            const phone = row.querySelector('.inp-phone').value.trim();
            const email = row.querySelector('.inp-email').value.trim();

            if (phone.length > 0) hasPhone = true;

            if(name || phone || email) { 
                contacts.push({ name, role, phone, email });
            }
        });

        if (!hasPhone) {
            openAlertModal("Il faut ajouter au moins un num√©ro de t√©l√©phone valide.");
            return;
        }

        const data = { id: id || 'cli-'+Date.now(), alert: alertVal, company, contact, address, status, notes, contacts, tarif };
